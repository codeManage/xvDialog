xvDialog.js一款轻量级的弹层插件

使用说明：

需要引入插件xvDialog.js/xvDialog.min.js

须要引入xvDialog.css 样式表，可以自定义自己想要的皮肤

本弹出层组件基于jquery库开发


使用方法如下：

$.xvDialog({
    index: 0,//弹出层的索引
    type: 'dialog',//弹出层的类型，tips
    drag: false,//是否开启拖拽功能，默认开启
    width:250,//弹出层的宽度
    width:150,//弹出层的高度
    closeAction:'hide',//弹出层的关闭类型，默认直接销毁，hide表示隐藏并不销毁
    title: false,//是否开启标题，默认开启
    closeBtn: false,//是否开启关闭按钮，默认开启
    icon: false,//是否开启图标，默认开启
    closeTimes:3000,//是否开启自动关闭，默认不开启
    iconType: 'waring', //选择图标类型 分别有，success、waring、error、info
    contentMsg: '我是xv-大前端！欢迎使用xvDialog欢迎使用欢迎使用欢迎使用欢迎使用欢迎使用欢迎使用欢迎使用欢迎使用'//弹出层的内容
    buttons: false,//是否显示弹层脚步的按钮，可以自定义，默认显示底部默认按钮，也可以自定义，如下：
    /*button: [
